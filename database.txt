-- =============================================================================
-- CRIMEPIOT AI DATABASE SCHEMAS (MySQL / MariaDB)
-- Contains both Node.js (Sequelize) and Django (Python) schemas.
-- =============================================================================

CREATE DATABASE IF NOT EXISTS `crimepilot` CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;
USE `crimepilot`;

-- -----------------------------------------------------------------------------
-- PART 1: Node.js (Sequelize ORM) Tables
-- -----------------------------------------------------------------------------

-- 1. Users Table
CREATE TABLE IF NOT EXISTS `users` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL,
  `email` VARCHAR(100) NOT NULL UNIQUE,
  `password` VARCHAR(255) NOT NULL,
  `role` ENUM('admin', 'officer', 'analyst', 'citizen') DEFAULT 'officer',
  `isActive` BOOLEAN DEFAULT TRUE,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 2. Locations Table (Jurisdictions & Stations)
CREATE TABLE IF NOT EXISTS `locations` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `state` VARCHAR(100) NOT NULL,
  `district` VARCHAR(100) NOT NULL,
  `city` VARCHAR(100) NOT NULL,
  `policeStation` VARCHAR(200) NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY `unique_location` (`state`, `district`, `city`, `policeStation`)
) ENGINE=InnoDB;

-- 3. Officers Profile Table
CREATE TABLE IF NOT EXISTS `officers` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userId` INT DEFAULT NULL,
  `badgeNo` VARCHAR(50) NOT NULL UNIQUE,
  `stationId` INT DEFAULT NULL,
  `contact` VARCHAR(20) NOT NULL,
  `profilePicture` VARCHAR(255) DEFAULT '',
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`userId`) REFERENCES `users` (`id`) ON DELETE CASCADE ON UPDATE CASCADE,
  FOREIGN KEY (`stationId`) REFERENCES `locations` (`id`) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 4. Analysts Profile Table
CREATE TABLE IF NOT EXISTS `analysts` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userId` INT DEFAULT NULL,
  `department` VARCHAR(100) NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`userId`) REFERENCES `users` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 5. Crime Categories Table
CREATE TABLE IF NOT EXISTS `crime_categories` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL UNIQUE,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 6. Crime Category Sections Table (Feature A: Legal Acts matching Categories)
CREATE TABLE IF NOT EXISTS `crime_category_sections` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `categoryId` INT NOT NULL,
  `act` VARCHAR(100) NOT NULL,
  `section` VARCHAR(100) NOT NULL,
  `description` TEXT NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`categoryId`) REFERENCES `crime_categories` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 7. Crimes Table
CREATE TABLE IF NOT EXISTS `crimes` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `crimeId` VARCHAR(30) UNIQUE NOT NULL,
  `categoryId` INT NOT NULL,
  `date` DATE NOT NULL,
  `time` VARCHAR(10) NOT NULL,
  `locationId` INT NOT NULL,
  `description` TEXT NOT NULL,
  `officerId` INT NOT NULL,
  `priority` ENUM('Low', 'Medium', 'High', 'Critical') DEFAULT 'Medium',
  `status` ENUM('Reported', 'Assigned', 'Under Investigation', 'Evidence Collected', 'Solved', 'Closed') DEFAULT 'Reported',
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`categoryId`) REFERENCES `crime_categories` (`id`) ON UPDATE CASCADE,
  FOREIGN KEY (`locationId`) REFERENCES `locations` (`id`) ON UPDATE CASCADE,
  FOREIGN KEY (`officerId`) REFERENCES `officers` (`id`) ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 8. Crime Notes Table (Updates timeline / notes log)
CREATE TABLE IF NOT EXISTS `crime_notes` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `crimeId` INT NOT NULL,
  `note` TEXT NOT NULL,
  `addedById` INT NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`crimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE,
  FOREIGN KEY (`addedById`) REFERENCES `users` (`id`) ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 9. Crime Selected Sections Table (Sections bounded to a case)
CREATE TABLE IF NOT EXISTS `crime_selected_sections` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `crimeId` INT NOT NULL,
  `act` VARCHAR(100) NOT NULL,
  `section` VARCHAR(100) NOT NULL,
  `description` TEXT NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`crimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 10. Evidences Table
CREATE TABLE IF NOT EXISTS `evidences` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `type` VARCHAR(100) NOT NULL,
  `description` TEXT NOT NULL,
  `collectionDate` DATETIME NOT NULL,
  `assignedOfficerId` INT DEFAULT NULL,
  `linkedCrimeId` INT DEFAULT NULL,
  `filePath` VARCHAR(255) DEFAULT '',
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`assignedOfficerId`) REFERENCES `officers` (`id`) ON DELETE SET NULL ON UPDATE CASCADE,
  FOREIGN KEY (`linkedCrimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 11. Suspects Table
CREATE TABLE IF NOT EXISTS `suspects` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL,
  `age` INT DEFAULT NULL,
  `gender` VARCHAR(10) DEFAULT '',
  `address` TEXT DEFAULT NULL,
  `photoPath` VARCHAR(255) DEFAULT '',
  `status` VARCHAR(50) DEFAULT 'Suspect',
  `linkedCrimeId` INT DEFAULT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`linkedCrimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 12. Suspects Previous Cases Join Table (Feature C: Similar suspects connection)
CREATE TABLE IF NOT EXISTS `suspect_previous_cases` (
  `suspectId` INT NOT NULL,
  `crimeId` INT NOT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`suspectId`, `crimeId`),
  FOREIGN KEY (`suspectId`) REFERENCES `suspects` (`id`) ON DELETE CASCADE ON UPDATE CASCADE,
  FOREIGN KEY (`crimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 13. Notifications Table
CREATE TABLE IF NOT EXISTS `notifications` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `type` VARCHAR(50) NOT NULL,
  `recipientId` INT NOT NULL,
  `message` TEXT NOT NULL,
  `read` BOOLEAN DEFAULT FALSE,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`recipientId`) REFERENCES `users` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 14. Audit Logs Table
CREATE TABLE IF NOT EXISTS `audit_logs` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userId` INT DEFAULT NULL,
  `action` VARCHAR(255) NOT NULL,
  `details` TEXT DEFAULT NULL,
  `method` VARCHAR(10) NOT NULL,
  `url` VARCHAR(255) NOT NULL,
  `ip` VARCHAR(45) NOT NULL,
  `timestamp` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`userId`) REFERENCES `users` (`id`) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 15. Victims Table
CREATE TABLE IF NOT EXISTS `victims` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL,
  `contact` VARCHAR(20) NOT NULL,
  `statement` TEXT NOT NULL,
  `evidenceReference` VARCHAR(255) DEFAULT '',
  `linkedCrimeId` INT DEFAULT NULL,
  `createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`linkedCrimeId`) REFERENCES `crimes` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;


-- -----------------------------------------------------------------------------
-- PART 2: Django (Python Backend) Django-managed Tables
-- These tables are managed under Django's Core and Authentication apps.
-- -----------------------------------------------------------------------------

-- 1. Django Custom Users Table
CREATE TABLE IF NOT EXISTS `authentication_customuser` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `password` VARCHAR(128) NOT NULL,
  `last_login` DATETIME DEFAULT NULL,
  `is_superuser` BOOLEAN NOT NULL DEFAULT FALSE,
  `email` VARCHAR(254) NOT NULL UNIQUE,
  `name` VARCHAR(150) NOT NULL,
  `role` VARCHAR(20) NOT NULL DEFAULT 'officer',
  `is_active` BOOLEAN NOT NULL DEFAULT TRUE,
  `is_staff` BOOLEAN NOT NULL DEFAULT FALSE,
  `profile_picture` VARCHAR(100) DEFAULT '',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 2. Django Officers Table
CREATE TABLE IF NOT EXISTS `authentication_officer` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `badge_no` VARCHAR(50) NOT NULL UNIQUE,
  `contact` VARCHAR(20) NOT NULL,
  `profile_picture` VARCHAR(100) DEFAULT '',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `station_id` INT DEFAULT NULL,
  `user_id` INT NOT NULL UNIQUE,
  FOREIGN KEY (`user_id`) REFERENCES `authentication_customuser` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 3. Django Analysts Table
CREATE TABLE IF NOT EXISTS `authentication_analyst` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `department` VARCHAR(100) NOT NULL,
  `profile_picture` VARCHAR(100) DEFAULT '',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `user_id` INT NOT NULL UNIQUE,
  FOREIGN KEY (`user_id`) REFERENCES `authentication_customuser` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 4. Django Citizens Table
CREATE TABLE IF NOT EXISTS `authentication_citizen` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `mobile` VARCHAR(20) NOT NULL,
  `dob` DATE DEFAULT NULL,
  `gender` VARCHAR(10) NOT NULL,
  `address` TEXT NOT NULL,
  `state` VARCHAR(100) NOT NULL,
  `city` VARCHAR(100) NOT NULL,
  `pincode` VARCHAR(10) NOT NULL,
  `identity_type` VARCHAR(50) NOT NULL,
  `identity_number` VARCHAR(100) NOT NULL,
  `identity_document` VARCHAR(100) DEFAULT '',
  `status` VARCHAR(20) NOT NULL DEFAULT 'Pending',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `user_id` INT NOT NULL UNIQUE,
  FOREIGN KEY (`user_id`) REFERENCES `authentication_customuser` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 5. Django Core Locations Table
CREATE TABLE IF NOT EXISTS `core_location` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `state` VARCHAR(100) NOT NULL,
  `district` VARCHAR(100) NOT NULL,
  `city` VARCHAR(100) NOT NULL,
  `police_station` VARCHAR(200) NOT NULL,
  UNIQUE KEY `unique_core_location` (`state`, `district`, `city`, `police_station`)
) ENGINE=InnoDB;

-- 6. Django Core Crime Categories Table
CREATE TABLE IF NOT EXISTS `core_crimecategory` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL UNIQUE,
  `sections` JSON NOT NULL
) ENGINE=InnoDB;

-- 7. Django Core Crimes Table
CREATE TABLE IF NOT EXISTS `core_crime` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `crime_id` VARCHAR(30) NOT NULL UNIQUE,
  `date` DATE NOT NULL,
  `time` VARCHAR(10) NOT NULL,
  `description` TEXT NOT NULL,
  `priority` VARCHAR(20) NOT NULL DEFAULT 'Medium',
  `status` VARCHAR(30) NOT NULL DEFAULT 'Reported',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `crime_category_id` INT NOT NULL,
  `location_id` INT NOT NULL,
  `officer_id` INT NOT NULL,
  FOREIGN KEY (`crime_category_id`) REFERENCES `core_crimecategory` (`id`) ON UPDATE CASCADE,
  FOREIGN KEY (`location_id`) REFERENCES `core_location` (`id`) ON UPDATE CASCADE,
  FOREIGN KEY (`officer_id`) REFERENCES `authentication_officer` (`id`) ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 8. Django Core Suspects Table
CREATE TABLE IF NOT EXISTS `core_suspect` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL,
  `age` INT DEFAULT NULL,
  `gender` VARCHAR(10) NOT NULL,
  `address` TEXT NOT NULL,
  `photo_path` VARCHAR(255) DEFAULT '',
  `status` VARCHAR(50) NOT NULL DEFAULT 'Suspect',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `linked_crime_id` INT NOT NULL,
  FOREIGN KEY (`linked_crime_id`) REFERENCES `core_crime` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 9. Django Core Victims Table
CREATE TABLE IF NOT EXISTS `core_victim` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `name` VARCHAR(100) NOT NULL,
  `contact` VARCHAR(20) NOT NULL,
  `statement` TEXT NOT NULL,
  `evidence_reference` VARCHAR(255) DEFAULT '',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `linked_crime_id` INT NOT NULL,
  FOREIGN KEY (`linked_crime_id`) REFERENCES `core_crime` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;

-- 10. Django Core Evidences Table
CREATE TABLE IF NOT EXISTS `core_evidence` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `type` VARCHAR(100) NOT NULL,
  `description` TEXT NOT NULL,
  `collection_date` DATETIME NOT NULL,
  `file_path` VARCHAR(255) DEFAULT '',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `assigned_officer_id` INT NOT NULL,
  `linked_crime_id` INT NOT NULL,
  FOREIGN KEY (`assigned_officer_id`) REFERENCES `authentication_officer` (`id`) ON UPDATE CASCADE,
  FOREIGN KEY (`linked_crime_id`) REFERENCES `core_crime` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;
